\newcommand{\limxa}{\lim\limits_{x\to a}}
\newcommand{\intab}{\int_{a}^{b}}
\newcommand{\iintD}{\iint\limits_{D}}
\newcommand{\ointC}{\oint\limits_{C}}

% 4. 矩阵与向量
